@app.route("/wishlist") def wishlist(): if "customer_email" not in session: return redirect("/customer-login") conn = sqlite3.connect("bookings.db") conn.row_factory = sqlite3.Row cursor = conn.cursor() cursor.execute(""" SELECT providers.* FROM providers JOIN wishlist ON providers.id=wishlist.provider_id WHERE wishlist.customer_email=? """,(session["customer_email"],)) providers = cursor.fetchall() conn.close() return render_template( "wishlist.html", providers=providers )
